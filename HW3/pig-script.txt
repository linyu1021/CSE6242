bigrams = LOAD 's3://cse6242-2017fall-aws-big/*' AS (bigram:chararray, year:int, num:int, books:int);filtered = FILTER bigrams BY (num >= (double)200) AND (books >= (double)10);group_by_bigram = GROUP filtered BY bigram;sum_by = FOREACH group_by_bigram {sum_num = SUM(filtered.num); sum_books = SUM(filtered.books);GENERATE group, (double)(sum_num)/(double)(sum_books) as sum;}order_by_sum = ORDER sum_by BY sum DESC, group ASC;result = LIMIT order_by_sum 10;STORE result INTO 's3://cse6242-lyao43/output-big' using PigStorage('\t');
